def self.from_raw_json(raw_json)
  parsed = JSON.parse(raw_json)
  new(
    result_type: 'CatletConfig',
    configuration: parsed.dig('result', 'configuration')
  )
rescue JSON::ParserError
  new(result_type: 'CatletConfig', configuration: nil)
end
